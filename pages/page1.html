var lines = [];
var circs = [];
var lineAng = 0;
var circAng = 0;
var rad = 100;

function setup() {
  createCanvas(400, 400);
  angleMode(DEGREES);
  colorMode(HSB)
  for (var i = 0; i < 26; i++) {
    lines[i] = new Lines(lineAng, rad);
    lineAng += 7.5;
  }
  for (var i = 0; i < 26; i++) {
    circs[i] = new Circs(circAng);
    circAng += 7.5;
    var n = circs[i].myFromAngle(circAng, 200);
    print("hey 2" + n.x);
  }
}

function draw() {
  background(12, 2, 70);
  translate(width / 2, height / 2);

  for (var i = 0; i < 26; i++) {
    lines[i].display();
  }
  for (var i = 0; i < 26; i++) {
    circs[i].display();
    circs[i].move();
  }
}

class Circs {
  constructor(angle) {
    this.angle = angle;
    this.rad2 = 100;
    this.fr = p5.Vector.fromAngle(radians(this.angle), this.rad2 * 2);
    this.vx = this.fr.x;
    this.vy = this.fr.y;
    this.r=random(100, 360);
    this.g=100;
    this.b=100;
    print("hey " + this.fr.x);
  }
  display() {
    push();
    fill(this.r, this.g, this.b);
    this.x2 = map(sin(this.angle), -1, 1, -this.vx, this.vx);
    this.y2 = map(sin(this.angle), -1, 1, -this.vy, this.vy);
    ellipse(this.x2, this.y2, 20, 20);
    pop();
  }
  move() {
    this.angle -= 0.5;
  }

  myFromAngle(angle, distance) {
    //Get SOH
    var op = sin(angle) * distance;
    //Get CAH
    var ad = cos(angle) * distance;
    //Add to old Vector
    var newVect = createVector(ad, op);
    return newVect;
  }
}

class Lines {
  constructor(angle, rad) {
    this.angle = angle;
    this.rad = rad;
    this.fr = p5.Vector.fromAngle(radians(this.angle), this.rad * 2);
    this.vx = this.fr.x;
    this.vy = this.fr.y;
  }
  display() {
    push();
    noFill();
    strokeWeight(3);
    stroke(0);
    line(0, 0, -this.vx, -this.vy);
    stroke(0);
    line(0, 0, this.vx, this.vy);
    pop();
  }
}

    function playAgain() {

    function makeCircles() {
      for (var i = 0; i < circles.length; i++) {
        var r = random(2, 8);
        x1 = cos(random(angle2)) * floor(random(200));
        y1 = sin(random(angle2)) * floor(random(200));
        angle2 += 0.1;
        if (abs(x1 * x1) + abs(y1 * y1) < 22500) {
          var newCircle = new Circle(x1, y1, r);
          for (var j = 0; j < circles.length; j++) {
            var other = circles[j];
            var d = dist(newCircle.x, newCircle.y, other.x, other.y);
            if (d < other.r + 2) {
              newCircle = undefined;
              break;
            }
          }
          if (newCircle) {
            circles.push(newCircle);
          }
        }
      }
    }
    
    function overlap() {
      for (var i = 0; i < circles.length; i++) {
        var c = circles[i];
        c.display();
        if (c.growing) {
          c.grow();
          // Does it overlap any previous circles?
          for (var j = 0; j < circles.length; j++) {
            var other = circles[j];
            if (other != c) {
              var d = dist(c.x, c.y, other.x, other.y);
              if (d - 1 < c.r + other.r) {
                c.growing = false;
              }
            }
          }
          // Is it stuck to an edge?
          if (c.growing) {
            c.growing = !c.edges();
          }
        }
      }
    }
    
    class Circle {
      constructor(x, y, r) {
        this.growing = true;
        this.x = x;
        this.y = y;
        this.r = r;
      }
      display() {
        noFill();
        stroke(255, 0, 0);
        strokeWeight(2);
        ellipse(this.x, this.y, this.r, this.r);
      }
      edges() {
        return (
          this.r > width - this.x ||
          this.r > this.x ||
          this.r > height - this.y ||
          this.r > this.y
        );
      }
      grow() {
        this.r += 0.5;
      }
    }
    class Pt {
      constructor() {
        this.x = x;
        this.y = y;
      }
      display() {
        stroke(0);
        strokeWeight(4);
        point(this.x, this.y);
      }
    }
